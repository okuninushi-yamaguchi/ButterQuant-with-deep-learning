import React, { useState, useEffect } from 'react';
import { useTranslation } from 'react-i18next';

interface RankingItem {
  rank: number;
  ticker: string;
  company_name: string;
  strategy_type: string;
  score: number;
  market_price: number;
  analysis_time: string;
}

interface RankingsData {
  last_updated: string;
  total_stocks: number;
  rankings: RankingItem[];
}

interface ScanStatus {
  status: 'no_data' | 'in_progress' | 'completed';
  date?: string;
  progress?: number;
  total?: number;
}

export const StrategyRank: React.FC = () => {
  const { t } = useTranslation();
  const [rankings, setRankings] = useState<RankingsData | null>(null);
  const [scanStatus, setScanStatus] = useState<ScanStatus | null>(null);
  const [displayCount, setDisplayCount] = useState(20);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  // 获取排名数据
  const fetchRankings = async () => {
    try {
      setLoading(true);
      const response = await fetch('/api/rankings/top20');
      
      if (!response.ok) {
        throw new Error('Rankings not available yet');
      }
      
      const data = await response.json();
      setRankings(data);
      setError(null);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to fetch rankings');
      console.error('Error fetching rankings:', err);
    } finally {
      setLoading(false);
    }
  };

  // 获取扫描状态
  const fetchScanStatus = async () => {
    try {
      const response = await fetch('/api/rankings/status');
      const data = await response.json();
      setScanStatus(data);
    } catch (err) {
      console.error('Error fetching scan status:', err);
    }
  };

  // 初始加载
  useEffect(() => {
    fetchRankings();
    fetchScanStatus();
    
    // 每30秒更新一次状态
    const interval = setInterval(fetchScanStatus, 30000);
    return () => clearInterval(interval);
  }, []);

  // 加载更多
  const loadMore = async () => {
    if (!rankings) return;
    
    try {
      const response = await fetch('/api/rankings');
      const data = await response.json();
      setRankings(data);
      setDisplayCount(prev => Math.min(prev + 20, data.total_stocks));
    } catch (err) {
      console.error('Error loading more:', err);
    }
  };

  // 格式化时间
  const formatTime = (isoString: string) => {
    const date = new Date(isoString);
    return date.toLocaleString('en-US', {
      month: 'short',
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit',
      timeZone: 'Asia/Tokyo',
      timeZoneName: 'short'
    });
  };

  // 获取策略类型的颜色
  const getStrategyColor = (strategyType: string) => {
    const colors: Record<string, string> = {
      'Long Call Butterfly': 'text-blue-600 bg-blue-50',
      'Long Put Butterfly': 'text-red-600 bg-red-50',
      'Iron Butterfly': 'text-purple-600 bg-purple-50',
      'Unknown': 'text-gray-600 bg-gray-50'
    };
    return colors[strategyType] || colors['Unknown'];
  };

  // 计算数据新鲜度
  const getDataFreshness = () => {
    if (!rankings) return null;
    
    const lastUpdate = new Date(rankings.last_updated);
    const now = new Date();
    const hoursAgo = Math.floor((now.getTime() - lastUpdate.getTime()) / (1000 * 60 * 60));
    
    if (hoursAgo < 1) {
      return { text: 'Fresh data', color: 'text-green-600' };
    } else if (hoursAgo < 24) {
      return { text: `${hoursAgo}h ago`, color: 'text-yellow-600' };
    } else {
      return { text: `${Math.floor(hoursAgo / 24)}d ago`, color: 'text-red-600' };
    }
  };

  const freshness = getDataFreshness();

  return (
    <div className="container mx-auto px-4 py-8">
      {/* 标题和状态 */}
      <div className="mb-6">
        <div className="flex justify-between items-center mb-4">
          <h1 className="text-3xl font-bold text-gray-800">
            {t('strategy_rank.title', 'Strategy Rankings')}
          </h1>
          
          {freshness && (
            <div className="flex items-center gap-2">
              <span className="text-sm text-gray-600">Last updated:</span>
              <span className={`text-sm font-medium ${freshness.color}`}>
                {rankings && formatTime(rankings.last_updated)}
              </span>
            </div>
          )}
        </div>

        {/* 扫描状态提示 */}
        {scanStatus && scanStatus.status === 'in_progress' && (
          <div className="bg-blue-50 border border-blue-200 rounded-lg p-4 mb-4">
            <div className="flex items-center justify-between">
              <div className="flex items-center gap-3">
                <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-blue-600"></div>
                <span className="text-blue-800 font-medium">
                  Scanning in progress...
                </span>
              </div>
              <span className="text-blue-600 text-sm">
                {scanStatus.progress} / {scanStatus.total} stocks
              </span>
            </div>
            {scanStatus.progress && scanStatus.total && (
              <div className="mt-3 w-full bg-blue-200 rounded-full h-2">
                <div 
                  className="bg-blue-600 h-2 rounded-full transition-all duration-300"
                  style={{ width: `${(scanStatus.progress / scanStatus.total) * 100}%` }}
                ></div>
              </div>
            )}
          </div>
        )}

        {/* 说明 */}
        <p className="text-gray-600">
          {t('strategy_rank.description', 'Rankings are updated daily at 18:00 JST covering Nasdaq 100 and S&P 500 stocks.')}
        </p>
      </div>

      {/* 加载状态 */}
      {loading && !rankings && (
        <div className="flex justify-center items-center py-12">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600"></div>
        </div>
      )}

      {/* 错误状态 */}
      {error && (
        <div className="bg-red-50 border border-red-200 rounded-lg p-4 text-red-800">
          <p className="font-medium">Error loading rankings</p>
          <p className="text-sm mt-1">{error}</p>
          <button 
            onClick={fetchRankings}
            className="mt-3 px-4 py-2 bg-red-600 text-white rounded hover:bg-red-700 transition"
          >
            Retry
          </button>
        </div>
      )}

      {/* 排名表格 */}
      {!loading && rankings && (
        <>
          <div className="bg-white rounded-lg shadow-md overflow-hidden">
            <div className="overflow-x-auto">
              <table className="min-w-full divide-y divide-gray-200">
                <thead className="bg-gray-50">
                  <tr>
                    <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                      Rank
                    </th>
                    <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                      Ticker
                    </th>
                    <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                      Company
                    </th>
                    <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                      Strategy
                    </th>
                    <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                      Score
                    </th>
                    <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                      Price
                    </th>
                  </tr>
                </thead>
                <tbody className="bg-white divide-y divide-gray-200">
                  {rankings.rankings.slice(0, displayCount).map((item) => (
                    <tr 
                      key={item.ticker}
                      className="hover:bg-gray-50 transition cursor-pointer"
                      onClick={() => window.location.href = `/?ticker=${item.ticker}`}
                    >
                      <td className="px-6 py-4 whitespace-nowrap">
                        <span className={`text-lg font-bold ${
                          item.rank <= 3 ? 'text-yellow-500' : 'text-gray-600'
                        }`}>
                          #{item.rank}
                        </span>
                      </td>
                      <td className="px-6 py-4 whitespace-nowrap">
                        <span className="text-sm font-medium text-blue-600">
                          {item.ticker}
                        </span>
                      </td>
                      <td className="px-6 py-4">
                        <span className="text-sm text-gray-900">
                          {item.company_name}
                        </span>
                      </td>
                      <td className="px-6 py-4 whitespace-nowrap">
                        <span className={`px-3 py-1 rounded-full text-xs font-medium ${getStrategyColor(item.strategy_type)}`}>
                          {item.strategy_type}
                        </span>
                      </td>
                      <td className="px-6 py-4 whitespace-nowrap">
                        <div className="flex items-center">
                          <span className="text-sm font-semibold text-gray-900">
                            {item.score.toFixed(4)}
                          </span>
                          <div className="ml-2 w-24 bg-gray-200 rounded-full h-2">
                            <div 
                              className="bg-green-500 h-2 rounded-full"
                              style={{ width: `${Math.min(item.score * 100, 100)}%` }}
                            ></div>
                          </div>
                        </div>
                      </td>
                      <td className="px-6 py-4 whitespace-nowrap">
                        <span className="text-sm text-gray-900">
                          ${item.market_price.toFixed(2)}
                        </span>
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          </div>

          {/* 加载更多按钮 */}
          {displayCount < rankings.total_stocks && (
            <div className="mt-6 text-center">
              <button
                onClick={loadMore}
                className="px-6 py-3 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition font-medium"
              >
                {t('strategy_rank.load_more', 'Load More')} 
                ({displayCount} / {rankings.total_stocks})
              </button>
            </div>
          )}

          {/* 统计信息 */}
          <div className="mt-6 grid grid-cols-1 md:grid-cols-3 gap-4">
            <div className="bg-gradient-to-br from-blue-50 to-blue-100 rounded-lg p-4">
              <p className="text-sm text-blue-600 font-medium">Total Stocks</p>
              <p className="text-2xl font-bold text-blue-900">{rankings.total_stocks}</p>
            </div>
            <div className="bg-gradient-to-br from-green-50 to-green-100 rounded-lg p-4">
              <p className="text-sm text-green-600 font-medium">Displayed</p>
              <p className="text-2xl font-bold text-green-900">{displayCount}</p>
            </div>
            <div className="bg-gradient-to-br from-purple-50 to-purple-100 rounded-lg p-4">
              <p className="text-sm text-purple-600 font-medium">Next Update</p>
              <p className="text-2xl font-bold text-purple-900">18:00 JST</p>
            </div>
          </div>
        </>
      )}
    </div>
  );
};

export default StrategyRank;
