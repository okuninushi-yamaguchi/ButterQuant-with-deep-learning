# docker-compose.yml - 放在项目根目录

version: '3.8'

services:
  # ==================== Redis 缓存 ====================
  redis:
    image: redis:7-alpine
    container_name: butterquant_redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    command: redis-server --appendonly yes --maxmemory 512mb --maxmemory-policy allkeys-lru
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 3

  # ==================== Flask 后端 ====================
  backend:
    build: ./backend
    container_name: butterquant_backend
    ports:
      - "5000:5000"
    environment:
      - REDIS_HOST=redis
      - REDIS_PORT=6379
      - CACHE_TTL=300  # 5分钟缓存
      - WORKERS=4      # Gunicorn worker 数量
    volumes:
      - ./backend:/app
      - ./backend/data:/app/data  # 挂载数据库目录
    depends_on:
      redis:
        condition: service_healthy
    restart: unless-stopped
    command: >
      gunicorn 
      --bind 0.0.0.0:5000 
      --workers 4 
      --worker-class gevent 
      --worker-connections 1000 
      --timeout 120 
      --access-logfile - 
      --error-logfile - 
      app:app
